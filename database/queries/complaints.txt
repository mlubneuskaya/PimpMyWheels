CREATE OR REPLACE TABLE complaints (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY COMMENT 'Id reklamacji',
    client_id INT UNSIGNED COMMENT 'Klient który zgłosił',
    employee_id INT UNSIGNED COMMENT 'Pracownik który przyjął',
    open_date TIMESTAMP NOT NULL COMMENT 'Data przyjęcia',
    closure_date TIMESTAMP COMMENT 'Data zamknięcia reklamacji',
    car_id INT UNSIGNED COMMENT 'Pojazd którego dotyczy',
    description CHAR(100) NOT NULL,
    cost DECIMAL(8,2) NOT NULL COMMENT 'Koszt który poniósł warsztat',
    FOREIGN KEY (client_id) REFERENCES clients(id),
    FOREIGN KEY (employee_id) REFERENCES employees(id),
    FOREIGN KEY (car_id) REFERENCES equipment(id)
) ENGINE=InnoDB CHARACTER SET=utf8mb4 COLLATE=utf8mb4_polish_ci;
